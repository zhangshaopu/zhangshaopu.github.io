递归公式的思想：

dfs深搜：（状态）先纵后横，是对解的一种试探

  1. 数独： dfs（）是典型的深搜代码框架。

  2. 脑子里面时刻要有一副二叉树的图像，先往深处搜索（即转移到下一个状态），遇到不满足条件了，返回上一个状态（即回溯），随即进入平行状态。（即二叉树的兄弟结点）

  3. dfs（）的参数用于表明状态。在数独里面就是x，y。用于表明table(x,y)。

  4. dfs（）函数编写思路：

    • 如果满足需要处理的条件，对这个位置进行操作。
      1. 先用check()判断 输入能否某值，如果能则赋值，进行下一个位置深搜；不能就试下一个；
      2. 回溯：返回上一个状态；在上一个状态进入平行状态。
    • 否则，继续找下一个需要处理的位置。eg:比如在数独里，开始位置就是（0，0），倘若这个空已经填写了，则继续找下一个需要填写的空进行深搜。

  5. java Scanner 输入问题：

    nextInt(): it only reads the int value, nextInt() places the cursor in the same line after reading the input.将空格看作是两个输入的数据的间隔， nextInt()只读取数值，剩下"\n"还没有读取，并将cursor放在本行中。

    next(): read the input only till the space. It can't read two words separated by space. Also, next() places the cursor in the same line after reading the input.

    nextLine(): reads input including space between the words (that is, it reads till the end of line \n). Once the input is read, nextLine() positions the cursor in the next line.读取内容并自动换行,不将空格看做是两个字符串的间隔，而是看作字符串的一部分，返回时，它作为String类型一并返回.

  6. char 和 int类型的转换：

     table[i][j]=line.charAt(j)-'0';

  7. 注意状态回溯要写在进入平行状态dfs之前。

package train;
import java.util.Scanner;
/**
 005300000
 800000020
 070010500
 400005300
 010070006
 003200080
 060500009
 004000030
 000009700
 */

public class shudu {
    static int[][] table = new int[9][9];
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);

        for(int i=0;i<9;i++){
            String line=sc.nextLine();
            for(int j=0;j<9;j++){
                table[i][j]=line.charAt(j)-'0';
            }
        }
        //System.out.println(table[1][1]);
        print();
        dfs(table,0,0);
    }
    private static void dfs(int[][] table, int x, int y) {
        if(x==9){
            print();
            System.exit(0);//另有情况用return
        }
        if(table[x][y]==0){
            //选1-9之间符合条件数字填到x，y位置上
            for(int i=1;i<10;i++){//进入下一个状态
                boolean res = check(table,x,y,i);//剪枝
                if(res) {
                    table[x][y] = i;
                    dfs(table, x + (y + 1) / 9, (y + 1) % 9);//纵深，深搜，进入到下一个状态；顺序为：先行后列
                }
            }
            //回溯，返回上一个状态
            table[x][y] =0;
        }
        else{
            //继续找下一个需要处理的位置
            dfs(table,x+(y+1)/9,(y+1)%9);
        }

    }
    private static void print() {
        for(int i=0;i<9;i++)
        {
            for(int j=0;j<9;j++)
            {
                System.out.print(table[i][j]);
            }
            System.out.println();
        }
    }
    //检查满足（1）每一行、（2）每一列、（3）每一个同色九宫内的数字均含1-9，不重复。三个条件
    private static boolean check(int[][] table, int x, int y, int value) {
        //检查同行同列
        for(int i=0;i<9;i++){
            if(value==table[x][i]) return false;
            if(value==table[i][y]) return false;
        }
        //检查小九宫格
        for(int l = (x/3)*3;l<(x/3+1)*3;l++){//行
            for(int m = (y/3)*3;m<(y/3+1)*3;m++){//列
                if(table[l][m]==value) return false;
            }
        }
        return true;
    }
}

package train;

import java.util.ArrayList;
import java.util.Scanner;
/*n=4
* a={1,2,4,7}
* k=13
* */
public class bufenhe {
    static int kk;
    public static void main(String[] args) {
        Scanner sc =new Scanner(System.in);
        int n = sc.nextInt();
        int[] a = new int[n];
        for(int i=0;i<n;i++){
            a[i]=sc.nextInt();
        }
        int k =sc.nextInt();
        kk=k;
        dfs_bufenhe(a,k,0,new ArrayList<Integer>());
    }

    private static void dfs_bufenhe(int[] a, int k,int cur,ArrayList<Integer> ints) {//变化的k和cur都是表示状态改变
        if(k==0&&cur<=a.length){
            System.out.print("yes("+kk+"=");
            int size=ints.size();
            for(int i=0;i<size;i++){
                System.out.print(ints.get(i)+(i==(size-1)?"":"+"));
                System.out.print(")");
            }

        }
        if(k<0||cur==a.length) return;//回退的条件
        if(k>0||cur<a.length) {//这一行可以注释
            ints.add(a[cur]);
            dfs_bufenhe(a, k - a[cur], cur + 1,ints);//要第cur个元素的值，进入下一个状态

            //回溯
            int index = ints.size()-1;
            ints.remove(index);
            dfs_bufenhe(a,k,cur+1,ints);//不要第cur个元素的值，即进入平行状态
        }
    }
}

package second;
/*w:表示水
  .:表示陆地
n=10
m=12
W........WW.
.WWW.....WWW
....WW...WW.
.........WW.
.........W..
..W......W..
.W.W.....WW.
W.W.W.....W.
.W.W......W.
..W.......W.
 */
import java.util.Scanner;

public class shuiwa {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        int n=sc.nextInt();
        int m=sc.nextInt();
        char[][] a=new char[n][];
        int count=0;
        for (int i = 0; i < n; i++) {
            a[i] = sc.next().toCharArray();
        }
        for (int i = 0; i < n; i++) {
            for (int j = 0; j < m; j++) {
                if(a[i][j]=='W') {
                    dfs(a,i,j);
                    count++;
                }
            }
        }
        System.out.println(count);
    }
    private static void dfs(char[][] a, int i, int j) {
        a[i][j] = '.';

        for (int k = -1; k < 2; k++) {//-1 0 1
            for (int l = -1; l < 2; l++) {//-1 0 1
                if(k==0&&l==0)continue;//排除自己

                if(i+k>=0&&i+k<=9&&j+l>=0&&j+l<=11) {//限制边界
                    if (a[i + k][j + l] == 'W')
                        dfs(a, i + k, j + l);
                }
            }
        }
    }
}

notes:

  1. 注意八个方向的写法，用两层循环 3*3 -1个数表示八个方向。
  2. 在这个dfs中没有用到回溯。因为假如用到回溯，还原回到遍历过的水坑状态，那么在它进入下一个状态时可能还是一开始的状态，就会出现死循环。所以采用把遍历过的水坑都变成陆地。
  3. next()和nextline()的区别；

/*
每行每列每条对角线上都只有一个棋子，求其摆放方法数
 */
public class nqueens {
    static int n=4;
    static int count = 0;

    public static void main(String[] args) {

        int[][] table = new int[n][n];
        dfs(0, table);
        System.out.println(count);
    }

    private static void dfs(int row, int[][] table) {
        if (row == n) {
            count++;
            return;//很灵性
        }
        for (int i = 0; i < n; i++) {
            //剪枝 行 列 对角线上都没有
            if (check(row, i, table)) {
                table[row][i] = 1;
                dfs(row + 1, table);
                table[row][i] = 0;
            }
        }
    }

    private static boolean check(int row, int col, int[][] table) {
        //其实不用对行进行判断，只需对列和对角线
        for (int i = 0; i < n; i++) {//对列
            if (table[i][col] == 1)
                return false;
        }
        for (int i = row,j=col; i>=0&&j>=0; i--,j--) {
            if(table[i][j]==1)
                return false;
        }
        for (int i = row,j=col; i>=0&&j<n; i--,j++) {
            if(table[i][j]==1)
                return false;
        }
        return true;
    }
}

notes:

  1. 对于对角线上的搜索
  2. for循环的逻辑问题，表达式2的意思是：当i/j 在上一次循环里 ++/-- 后，先判断是否满足表达式二，再进行下一次循环。

package forth;

import java.util.Scanner;

/**
 * @author zsp
 * @version v 0.1 2020/2/26 16:33
 */
public class sushuhuan {
    static int flag;
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        int n = sc.nextInt();
        int[] r = new int[n+1];
        r[1] = 1;
        dfs(2,r,n);
        if(flag!=1)
            System.out.println("no solution");
    }
    private static void dfs(int cur, int[] r,int n) {
        if(cur==n+1 && sushu(r[1]+r[n])){
            flag =1;
            print( r );
            return;
        }

        for (int i = 2; i <=n; i++) {
            if(check(r,i,cur))//1.与上一个数相加是素数 2.且这个数不在数组里面
            {
                r[cur] = i;
               dfs(cur+1,r,n);
               r[cur] = 0;
            }
        }

    }

    private static void print(int[] a) {
        for(int i=1;i<a.length;i++)
            System.out.print(a[i]+" ");
        System.out.println();
    }

    private static boolean check(int[] r,int val ,int cur) {
        int sum = val+r[cur-1];
        if(!sushu(sum))
            return false;
        for (int i = 1; i < cur; i++) {
            if(r[i]==val)
                return false;
        }

        return true;
    }

    private static boolean sushu(int temp) {
        for (int i = 2; i < temp; i++) {
            if(temp%i==0)
                return false;
        }
        return true;
    }
}
